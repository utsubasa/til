def yakusuu():
    for i in range(1,line):
        if line % i == 0:
            yakusuu_list.append(i)
    yakusuu_str = list(map(int,yakusuu_list))
    ans = sum(yakusuu_str)
    if ans == line:
        print("perfect")
    elif ans == abs(line -1):
        print("nearly")
    else:
        print("neither")
